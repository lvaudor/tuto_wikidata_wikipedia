get_genre_and_actors=function(film_id){
  film_id=paste0("<",film_id,">")
  result=spq_init() %>%
      spq_set(film= film_id) %>%
      spq_add("?film wdt:P136 ?genre") %>%
      spq_add("?film wdt:P161 ?actor",.required=FALSE) %>%
      spq_label(genre, actor) %>%
      spq_select(-film) %>%
      spq_perform()
  return(result)
}
